## Pre-processing

response = "label"
y = df[response]

quan_preds = ["age", "fnlwgt", "capital-gain", "capital-loss", "hours-per-week"]
quan_df = df[quan_preds]
newdf = df.drop(columns = quan_preds)
newdf = newdf.drop(columns = response)
qual_df = pd.get_dummies(newdf)
X = pd.concat([quan_df, qual_df], axis=1)

X.shape

(32561, 108)

# Simple Tree

clf = DecisionTreeClassifier(max_depth = 3, random_state = 42)
clf = clf.fit(X, y)
accuracy_CART = cross_val_score(clf, X, y, cv = 10).mean()

# Bagging

clf_bagging = BaggingClassifier(base_estimator=DecisionTreeClassifier(), n_estimators=10, random_state=0)
clf_bagging = clf_bagging.fit(X, y)
accuracy_bagging = cross_val_score(clf_bagging, X, y, cv = 10).mean()

# Bagging + SU

np_X = np.array(X)
np_y = np.array(y)
scores = su_measure(np_X, np_y)
data = {"Feature": list(X.columns.values), "Score": list(scores)}  
su_scores_df = pd.DataFrame(data) 
a = su_scores_df.sort_values(by=['Score'], ascending=False)
a[a["Score"] > .9].shape

(50, 2)

su_features = su_scores_df.sort_values(by=['Score'], ascending=False)[0:50]["Feature"].tolist()
X_SU = X[su_features]
clf_SU = BaggingClassifier(base_estimator=DecisionTreeClassifier(), n_estimators=10, random_state=0)
clf_SU = clf_SU.fit(X_SU, y)
accuracy_SU = cross_val_score(clf_SU, X_SU, y, cv = 10).mean()

# Bagging + SVM

svm_features = svm.LinearSVC().fit(X,y)
weighted_features = sorted(zip(svm_features.coef_[0],svm_features.feature_names_in_), key=lambda sublist: abs(sublist[0]))[::-1]
cols = [j for i,j in weighted_features[:50]]
X_SVM = X[cols]
clf_SVM = DecisionTreeClassifier(max_depth = 5)
clf_SVM = clf_SVM.fit(X_SVM,y)

accuracy_SVM = cross_val_score(clf_SVM, X_SVM, y, cv = 10).mean()
